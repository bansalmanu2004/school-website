} // Function to download Excel file function downloadExcel() { let tableData = [ ["Field", "Value"], ["School Name", document.getElementById("schoolName").value], ["School UDISE Code", document.getElementById("schoolCode").value], ["Price/Rate of Cooking Cost", document.getElementById("cookingCost").value], ["Total Students", document.getElementById("totalStudents").value], ["Opening Balance of Fruit", document.getElementById("openingFruit").value], ["Amount Received", document.getElementById("amountReceived").value], ["Number of Students Given Fruit", document.getElementById("studentsGivenFruit").value], ["Amount Spent in Month", document.getElementById("amountSpent").value] ]; let ws = XLSX.utils.aoa_to_sheet(tableData); let wb = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(wb, ws, "MDM Report"); XLSX.writeFile(wb, "MDM_Monthly_Report.xlsx"); } // Function to download PDF function downloadPDF() { const { jsPDF } = window.jspdf; let doc = new jsPDF(); doc.setFillColor(240, 240, 240); doc.rect(10, 10, 190, 280, 'F'); doc.setFontSize(18); doc.setTextColor(0, 51, 153); doc.text("MDM Monthly Report", 65, 20); let data = [ ["School Name:", document.getElementById("schoolName").value], ["School UDISE Code:", document.getElementById("schoolCode").value], ["Price/Rate of Cooking Cost:", document.getElementById("cookingCost").value], ["Total Students:", document.getElementById("totalStudents").value], ["Opening Balance of Fruit:", document.getElementById("openingFruit").value], ["Amount Received:", document.getElementById("amountReceived").value], ["Students Given Fruit:", document.getElementById("studentsGivenFruit").value], ["Amount Spent:", document.getElementById("amountSpent").value] ]; let y = 40; doc.setFontSize(12); doc.setTextColor(0, 0, 0); data.forEach(([key, value]) => { doc.text(`${key} ${value}`, 20, y); y += 10; }); doc.save("MDM_Monthly_Report.pdf"); }
